CREATE TABLE Registry (
	ID int,
	NAME varchar (50),
	RATING_PK int,
	AUTHOR_PK int, 
	URL varchar(255),
	BINARY_PK int,
	VERSION varchar (15),
	UPLOADED datetime,
	IS_EXTERNAL boolean,
	PRIMARY KEY (ID),
	FOREIGN KEY(RATING_PK) REFERENCES Rating(ID),
	FOREIGN KEY(AUTHOR_PK) REFERENCES Users(ID),
	FOREIGN KEY(BINARY_PK) REFERENCES Binaries(ID),
);
