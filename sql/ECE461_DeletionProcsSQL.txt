SQL STORED DELETION ENDPOINT PROCS
/reset (DELETE)
BEGIN CREATE PROCEDURE Reset_Registry
AS
	DELETE * FROM Registry
GO
END;

/package/{id} (DELETE)
BEGIN CREATE PROCEDURE Reset_Registry
AS
	DELETE FROM Registry
	WHERE ID = @@ID
GO
END;

/package/byName/{name} (DELETE)
BEGIN CREATE PROCEDURE Reset_Registry
AS
	DELETE FROM Registry
	WHERE NAME = @@name
GO
END;
